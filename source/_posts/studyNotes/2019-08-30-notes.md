代码：
```js
let data = {
  L1: [1,12,5,"L1"],
  B1: [1,12,5,"B1"],
  L3: [1,12,5,"L3"],
  L2: [1,12,5,"L2"]
};
let arr1 = [], arr2 = [];
for(let i in data) {
  arr1.push(i);
}
arr1.sort();
for(let i in arr1) {
  arr2.push(data[arr1[i]]);
}
console.log(arr2);
```
结果：
```
[ [ 1, 12, 5, 'B1' ],
  [ 1, 12, 5, 'L1' ],
  [ 1, 12, 5, 'L2' ],
  [ 1, 12, 5, 'L3' ] ]
```
